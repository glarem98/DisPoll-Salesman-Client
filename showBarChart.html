// Use Morris.Bar
Morris.Bar({
  element: 'graph',
  data: [
    {x: 'گزینه اول', y: 50},
    {x: 'گزینه دوم', y: 10},
    {x: 'گزینه سوم', y: 20},
    {x: 'گزینه چهارم', y: 32},
    {x: 'گزینه پنجم', y: 40}
  ],
  xkey: 'x',
  ykeys: ['y'],
  labels: ['Y'],
  barColors: function (row, series, type) {
    if (type === 'bar') {
      var red = Math.ceil(255 * row.y / this.ymax);
      return 'rgb(' + red + ',0,0)';
    }
    else {
      return '#000';
    }
  }
});
